// Quick Start

Up in 60 Seconds

Add Sourcegraph MCP to your favorite AI environment with a few lines of config.

{ "mcpServers": { "sourcegraph": { "command": "npx", "args": [ "-y", "@sourcegraph/mcp-server" ], "env": { "SRC_ENDPOINT": "https://sourcegraph.example.com", "SRC_ACCESS_TOKEN": "sgp_xxxxxxxxxxxxxxxxxxxx" } } } }
{ // .cursor/mcp.json "mcpServers": { "sourcegraph": { "command": "npx", "args": ["-y", "@sourcegraph/mcp-server"], "env": { "SRC_ENDPOINT": "https://sourcegraph.example.com", "SRC_ACCESS_TOKEN": "${env:SOURCEGRAPH_TOKEN}" } } } }
{ // settings.json — VS Code w/ Copilot Chat "github.copilot.chat.mcpServers": { "sourcegraph": { "command": "npx", "args": ["-y", "@sourcegraph/mcp-server"], "env": { "SRC_ENDPOINT": "https://sourcegraph.example.com", "SRC_ACCESS_TOKEN": "${env:SRC_ACCESS_TOKEN}" } } } }
